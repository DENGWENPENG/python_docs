```python

class Stu:
    """这是一个Stu的类"""

    def __init__(self,name,age,sex):

        self.name = name
        self.age = age
        self.sex = sex

    def get_stu_info(self):
        print(f"{self.name}的年龄是{self.age}，性别是{self.sex}")

if __name__ == "__main__":
    # 实例化
    s1 = Stu("鲸长",20,"男")
    s2 = Stu("Jack",20,"男")
    # 调用一个Stu的方法
    s1.get_stu_info()  
    s2.get_stu_info()
```
